```js
class Product {
    static count = 0;

    constructor(name, unitPrice, number) {
        this.name = name;
        this.unitPrice = unitPrice;
        this.number = number;
        Product.count++;
    }

    get totalPrice() {
        return this.unitPrice * this.number;
    }

    increase() {
        this.number++;
    }
}
```

基础写法:

```js
function Product(name, unitPrice, number) {
    this.name = name;
    this.unitPrice = unitPrice;
    this.number = number;
    Product.count++;
}

Product.count = 0;

Product.prototype.increase = function () {
    this.number++;
};
```

实现访问器:

```js
var Product = (function () {
    function Product(name, unitPrice, number) {
        if (Object.getPrototypeOf(this) !== Product.prototype) {
            throw new Error("Class must be called with new");
        }

        this.name = name;
        this.unitPrice = unitPrice;
        this.number = number;
        Product.count++;
    }

    Object.defineProperty(Product.prototype, "totalPrice", {
        get: function () {
            return this.unitPrice * this.number;
        },
    });

    Object.defineProperty(this, "totalPrice", {
        get: function () {
            return this.unitPrice * this.number;
        },
    });

    Product.count = 0;

    Product.prototype.increase = function () {
        this.number++;
    };

    return Product;
})();
```

不能通过 new 调用构造函数的原型上的方法:

```js
var Product = (function () {
    function Product(name, unitPrice, number) {
        if (Object.getPrototypeOf(this) !== Product.prototype) {
            throw new Error("Class must be called with new");
        }

        this.name = name;
        this.unitPrice = unitPrice;
        this.number = number;
        Product.count++;
    }

    Object.defineProperty(Product.prototype, "totalPrice", {
        get: function () {
            return this.unitPrice * this.number;
        },
        enumerable: false,
    });

    Object.defineProperty(this, "totalPrice", {
        get: function () {
            return this.unitPrice * this.number;
        },
        enumerable: false,
    });

    Product.count = 0;

    Object.defineProperty(Product.prototype, "increase", {
        value: function () {
            if (Object.getPrototypeOf(this) === Product.prototype.increase.prototype) {
                throw new Error("increase is not a constructor");
            }
            this.number++;
        },
        enumerable: false,
    });

    return Product;
})();
```
