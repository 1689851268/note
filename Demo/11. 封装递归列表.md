1.  实现 Recursive.vue:

```vue
<template>
    <ul class="recursive-list-container">
        <li v-for="(item, index) in list" :key="index">
            <span
                :class="['list-title', { active: item.isSelected }]"
                :style="{
                    paddingLeft: hierarchy * 2 + 'em',
                    fontSize: 30 - hierarchy * 2 + 'px',
                }"
                @click="handleClick(item)"
            >
                {{ item.name }}
            </span>
            <RecursiveList
                v-if="hasChildren(item.children)"
                :list="item.children"
                :hierarchy="hierarchy + 1"
                @update:list="handleClick"
            />
        </li>
    </ul>
</template>

<script>
export default {
    name: "RecursiveList", // 只要设置了 name 属性, 就可以在递归组件中使用自身
    props: {
        list: {
            type: Array,
            default: () => [],
            // demo: [ { name: 'xxx', isSelected?: true, children?: [ { name: 'yyy', isSelected?: false } ] } ]
        },
        hierarchy: {
            type: Number,
            default: 0,
        },
    },
    emits: ["update:list"],
    methods: {
        hasChildren(list) {
            return list && list.length > 0;
        },
        handleClick(item) {
            this.$emit("update:list", item);
        },
    },
};
</script>

<style lang="scss" scoped>
.list-title {
    cursor: pointer;
    margin: 10px 0;

    &.active {
        color: red;
    }

    &:hover {
        font-weight: 700;
    }
}
</style>
```

2.  使用 Recursive.vue:

```vue
<template>
    <div>
        <!-- 其他组件... -->
        <RecursiveList :list="list" @update:list="handleClick" />
    </div>
</template>

<script>
import RecursiveList from "@/components/RecursiveList.vue";

export default {
    components: { RecursiveList },
    data() {
        return {
            list: [
                {
                    name: "xxx",
                    isSelected: false,
                    children: [
                        {
                            name: "yyy",
                            isSelected: false,
                        },
                    ],
                },
                {
                    name: "zzz",
                    isSelected: false,
                    children: [
                        {
                            name: "aaa",
                            isSelected: false,
                            children: [
                                {
                                    name: "ccc",
                                    isSelected: true,
                                },
                            ],
                        },
                        {
                            name: "bbb",
                            isSelected: false,
                        },
                    ],
                },
            ],
        };
    },
    methods: {
        handleClick(item) {
            // 递归遍历 list, 将所有 item 的 isSelected 设置为 false
            const traverse = (list) => {
                list.forEach((item) => {
                    item.isSelected = false;
                    if (item.children && item.children.length > 0) {
                        traverse(item.children);
                    }
                });
            };
            traverse(this.list);
            // 将 item 的 isSelected 设置为 true
            item.isSelected = true;
        },
    },
};
</script>
```
