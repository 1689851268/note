# webpack.config.js

- webpack.config.js 是 webpack 的配置文件，用于指示 webpack 的运行
- 在 webpack 运行之前，会先读取 webpack.config.js 文件
- webpack 是基于 node.js 开发出来的打包工具，因此在配置文件 webpack.config.js 中，支持使用 node.js 相关的语法和模块

```js
const path = require("path");

// 配置 webpack
module.exports = {
    // 指定模式
    mode: "development", // 开发模式-development、 生产模式=production
    // 入口文件，默认为 ./src/index.js
    entry: "./src/index.js",
    // 输出文件配置
    output: {
        filename: "main.js", // 输出文件名，默认为 main.js
        path: path.resolve(__dirname, "dist"), // 输出路径，默认为 ./dist/
    },
    // 配置 loader
    module: {
        rules: [], // 设置匹配规则
    },
    // 配置 plugins
    plugins: [],
};
```



# loader 加载器

- 在开发过程中，webpack 默认只能处理 .js .json 文件
- 对于其他文件，需要使用 loader 加载器才可以对其进行构建

# 构建 .css

1. `pnpm i css-loader style-loader -D`
2. 配置 webpack.config.js

```js
const HtmlWebpackPlugin = require("html-webpack-plugin");
const htmlWebpackPlugin = new HtmlWebpackPlugin();

module.exports = {
    mode: "development",
    plugins: [htmlWebpackPlugin],
    module: {
        // test-匹配的文件类型、 use-要调用的 loader
        rules: [{ test: /\.css$/, use: ["style-loader", "css-loader"] }],
        // use 中的 loader 顺序是固定的，webpack 会 [从后往前] 使用 use 里面的 loader 对代码进行构建
        // 待所有的 loader 都构建完后，再由 webpack 进行最后的构建打包
    },
};
```

3. 创建 .css 文件

```css
/* index.css */
html,
body {
    margin: 0;
    padding: 0;
    height: 100%;
    background-color: cadetblue;
}
```

4. 在入口文件中，导入 .css 文件

```js
// index.js
import './index.css';
```

5. 执行脚本 `pnpm open`，可以在浏览器中看到，样式生效了

   执行脚本 `pnpm build`，查看打包出来的文件：index.html + main.js
   就是说，.css 文件被打包进 main.js 中了



# 构建 .less

1. `pnpm i less-loader less -D`
2. 配置 webpack.config.js

```js
const HtmlWebpackPlugin = require("html-webpack-plugin");
const htmlWebpackPlugin = new HtmlWebpackPlugin();

module.exports = {
    mode: "development",
    plugins: [htmlWebpackPlugin],
    module: {
        rules: [{ test: /\.css/, use: ["style-loader", "css-loader"] }],
        // 配置 less-loader
        // 因为 less 是 less-loader 的内置依赖项，所以需要下载但不需要配置
        rules: [{ test: /\.less$/, use: ["style-loader", "css-loader", "less-loader"] }],
    },
};
```

3. 创建 .css 文件

```css
/* index.less */
html {
    body {
        margin: 0;
        padding: 0;
        height: 100%;
        background-color: cadetblue;
    }
}
```

4. 在入口文件中，导入 .css 文件

```js
// index.js
import "./index.less";
```

5. 执行脚本 `pnpm open`，可以在浏览器中看到，样式生效了

   执行脚本 `pnpm build`，查看打包出来的文件：index.html + main.js
   就是说，.less 文件被打包进 main.js 中了



# 构建过于高级的 js 语法

webpack 只能处理一部分高级 JS 语法，对于那些处理不了的 JS 语法，需要使用 babel-loader 进行构建

1. `pnpm i babel-loader @babel/core @babel/piugin-proposal-class-properties -D`
2. 配置 webpack.config.js

```js
const HtmlWebpackPlugin = require("html-webpack-plugin");

module.exports = {
    mode: "development",
    plugins: [new HtmlWebpackPlugin()],
    module: {
        rules: [
            {
                test: /\.js$/,
                exclude: /node_module/, // 可以使用 exclude 设置需要排除的文件(夹)
                use: {
                    // 配置 babel-loader
                    // 因为 @babel/core 是 babel-loader 的内置依赖项，所以需要下载但不需要配置
                    loader: "babel-loader",
                    options: {
                        // 配置插件，插件 @babel/plugin-proposal-class-properties 可用于处理 class 中的高级语法
                        plugins: ["@babel/plugin-proposal-class-properties"],
                    },
                },
            },
        ],
    },
};
```

3. 在入口文件 ./src/index.js 中编写 JS 的高级语法

```js
class Person {
    // 通过 static 定义静态属性
    // webpack 无法处理 "静态属性" 这个高级语法
    static info = "Person info";
}
console.log(Person.info);
```

5. 执行脚本 `pnpm open`，可以在浏览器中看到，`Person.info` 被成功打印出来了

   执行脚本 `pnpm build`，查看打包出来的文件：index.html + main.js



# 构建图片资源

1. `pnpm install url-loader file-loader -D`
2. 配置 webpack.config.js

```js
const HtmlWebpackPlugin = require("html-webpack-plugin");
const htmlWebpackPlugin = new HtmlWebpackPlugin();

module.exports = {
    mode: "development",
    plugins: [htmlWebpackPlugin],
    module: {
        // 配置 url-loader；只配置一个 loader 时，use 值可以为 String 类型
        // 因为 file-loader 是 url-loader 的内置依赖项，所以需要下载但不需要配置
        rules: [{ test: /\.jpg|png|gif$/, use: "url-loader" }],
        rules: [{ test: /\.less$/, use: ["style-loader", "css-loader", "less-loader"] }],

        // 注意：rules 的顺序也不能随意改变，webpack 是按照 rules 的顺序 [从后往前] 进行构建的
        // 就是说，如果有逻辑顺序问题的话，先构建的 rules 应该放在后面；如果没有逻辑顺序问题，则可随意设置

        // 本例中，需要先构建 .less 文件，然后构建里面用到的图片资源。所以构建 .less 文件的 rules 需要放在后面
    },
};
```

3. 创建 index.less 文件，使用图片资源

```less
/* index.less */
html {
    body {
        margin: 0;
        padding: 0;
        height: 100%;
        background-color: cadetblue;
        background-image: url("./vue.jpg");
        background-repeat: no-repeat;
    }
}
```

4. 创建入口文件 index.js，引入样式资源

```js
// index.js
import "./index.less";
```

5. 执行脚本 `pnpm open`，可以在浏览器中看到，样式生效了

   执行脚本 `pnpm build`，查看打包出来的文件：index.html + main.js + xxx.jpg
   xxx 为 webpack 处理后的图片名称，为一串能唯一标志该图片的哈希值

> #### 配置对象 options

```js
const HtmlWebpackPlugin = require("html-webpack-plugin");

module.exports = {
    module: {
        mode: "development",
        plugins: [new HtmlWebpackPlugin()],
        rules: [
            {
                test: /\.less$/,
                use: ["style-loader", "css-loader", "less-loader"],
            },
            {
                test: /\.(jpg|png|gif)$/,
                // 设置 Object 形式的 use
                use: {
                    // 通过 loader 字段指定加载器
                    loader: "url-loader",

                    // 设置配置对象 options
                    options: {
                        // 图片大小 <= 8kb 时，就会被编码成 base64 格式，变成字符串形式
                        // 优点: 减少请求数量（减轻服务器压力）
                        // 缺点：图片体积会更大（文件请求速度更慢）
                        limit: 8 * 1024,

                        // 问题：默认处理不了 html 中 img 图片
                        // 解决办法：使用 html-loader

                        // 新问题：因为 url-loader 默认使用 es6 模块化解析，而 html-loader 引入图片是 commonjs
                        // 所以，解析时会出问题：[object Module]
                        // 解决办法：关闭 url-loader 的 es6 模块化，使用 commonjs 解析
                        esModule: false,

                        // 图片名字默认为其 hashId，比较长，我们可以给图片重命名
                        // [hash:10] - 取图片的 hash 的前 10 位
                        // [ext] - 取文件原来扩展名
                        name: "[hash:10].[ext]",
                        
                        // 将图片文件打包到 ./dist/image 目录下
                        // 默认是打包到 ./dist 目录下
                        outputPath: "image",
                    },
                },
            },
        ],
    },
};
```

如果配置项比较少，可以简写：

```js
const HtmlWebpackPlugin = require("html-webpack-plugin");

module.exports = {
    module: {
        mode: "development",
        plugins: [new HtmlWebpackPlugin()],
        rules: [
            {
                test: /\.less$/,
                use: ["style-loader", "css-loader", "less-loader"],
            },
            {
                test: /\.(jpg|png|gif)$/,
                // 设置 String 形式的 use;  参数直接配置在字符串中
                use: "url-loader?limit=8192",
            },
        ],
    },
};
```
